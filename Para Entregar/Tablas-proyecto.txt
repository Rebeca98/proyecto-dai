--- Creación de tablas

create table Usuario (
nombre  char(30),
password  char(20),
Uid     int identity(1,1) primary key not null);

create table Cancion(
nombre  char(30),
artista char(20),
album   char(50),
año     int,
precio  money,
picURL  char(1000),
Cid     int identity(1,1) primary key not null);

create table Compra(
Cid       int FOREIGN KEY REFERENCES Cancion,
Uid       int FOREIGN KEY REFERENCES Usuario,
fecha     date,
monto     money,
primary key (Uid, Cid)
)
